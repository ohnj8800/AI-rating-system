def square_num(n,m):
    result=0
    for i in range(n,m+1):
        if i**0.5%1==0:
            result+=1
    return result
while True:
    n,m=map(int,input().split())
    if n==m==0:
        break
    if n>m:
        print(square_num(m,n))
    else:
        print(square_num(n,m))
